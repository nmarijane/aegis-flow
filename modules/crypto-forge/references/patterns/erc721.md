# ERC-721 NFT Patterns

## Basic NFT

Minimal ERC-721 using OpenZeppelin with metadata URI support.

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/token/ERC721/extensions/ERC721URIStorage.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract BasicNFT is ERC721, ERC721URIStorage, Ownable {
    uint256 private _nextTokenId;

    constructor(string memory name, string memory symbol)
        ERC721(name, symbol) Ownable(msg.sender)
    {}

    function safeMint(address to, string memory uri) external onlyOwner {
        uint256 tokenId = _nextTokenId++;
        _safeMint(to, tokenId);
        _setTokenURI(tokenId, uri);
    }

    // Required overrides
    function tokenURI(uint256 tokenId) public view override(ERC721, ERC721URIStorage)
        returns (string memory)
    {
        return super.tokenURI(tokenId);
    }

    function supportsInterface(bytes4 interfaceId) public view override(ERC721, ERC721URIStorage)
        returns (bool)
    {
        return super.supportsInterface(interfaceId);
    }
}
```

### tokenURI Pattern

Two common approaches:

**Base URI + Token ID** (most common for generative collections):

```solidity
function _baseURI() internal pure override returns (string memory) {
    return "ipfs://QmXYZ.../";
}

// tokenURI(42) -> "ipfs://QmXYZ.../42"
```

**Per-token URI** (for unique 1/1 art):


## Minting

### Public Mint with Price

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract MintableNFT is ERC721, Ownable {
    uint256 public constant MINT_PRICE = 0.05 ether;
    uint256 public constant MAX_SUPPLY = 10000;
    uint256 public constant MAX_PER_WALLET = 5;

    uint256 private _nextTokenId;
    string private _baseTokenURI;

    mapping(address => uint256) public mintCount;

    constructor(string memory name, string memory symbol, string memory baseURI)
        ERC721(name, symbol) Ownable(msg.sender)
    {
        _baseTokenURI = baseURI;
    }

    function mint(uint256 quantity) external payable {
        require(quantity > 0 && quantity <= MAX_PER_WALLET, "invalid quantity");
        require(mintCount[msg.sender] + quantity <= MAX_PER_WALLET, "wallet limit");
        require(_nextTokenId + quantity <= MAX_SUPPLY, "sold out");
        require(msg.value >= MINT_PRICE * quantity, "insufficient payment");

        mintCount[msg.sender] += quantity;

        for (uint256 i = 0; i < quantity; i++) {
            _safeMint(msg.sender, _nextTokenId++);
        }
    }

    function withdraw() external onlyOwner {
        (bool success, ) = owner().call{value: address(this).balance}("");
        require(success, "withdraw failed");
    }

    function _baseURI() internal view override returns (string memory) {
        return _baseTokenURI;
    }

    function totalSupply() public view returns (uint256) {
        return _nextTokenId;
    }
}
```

### Whitelist Mint (Merkle Tree Proof)

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/utils/cryptography/MerkleProof.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract WhitelistNFT is ERC721, Ownable {
    bytes32 public merkleRoot;
    uint256 public constant WL_PRICE = 0.03 ether;
    uint256 public constant PUBLIC_PRICE = 0.05 ether;
    uint256 public constant MAX_SUPPLY = 5000;

    uint256 private _nextTokenId;
    mapping(address => bool) public whitelistClaimed;

    enum Phase { Closed, Whitelist, Public }
    Phase public currentPhase;

    constructor(string memory name, string memory symbol, bytes32 _merkleRoot)
        ERC721(name, symbol) Ownable(msg.sender)
    {
        merkleRoot = _merkleRoot;
    }

    function whitelistMint(bytes32[] calldata proof) external payable {
        require(currentPhase == Phase.Whitelist, "WL not active");
        require(!whitelistClaimed[msg.sender], "already claimed");
        require(_nextTokenId < MAX_SUPPLY, "sold out");
        require(msg.value >= WL_PRICE, "insufficient payment");

        // Verify Merkle proof
        bytes32 leaf = keccak256(abi.encodePacked(msg.sender));
        require(MerkleProof.verify(proof, merkleRoot, leaf), "invalid proof");

        whitelistClaimed[msg.sender] = true;
        _safeMint(msg.sender, _nextTokenId++);
    }

    function publicMint() external payable {
        require(currentPhase == Phase.Public, "public not active");
        require(_nextTokenId < MAX_SUPPLY, "sold out");
        require(msg.value >= PUBLIC_PRICE, "insufficient payment");

        _safeMint(msg.sender, _nextTokenId++);
    }

    function setPhase(Phase phase) external onlyOwner {
        currentPhase = phase;
    }

    function setMerkleRoot(bytes32 _merkleRoot) external onlyOwner {
        merkleRoot = _merkleRoot;
    }

    function withdraw() external onlyOwner {
        (bool success, ) = owner().call{value: address(this).balance}("");
        require(success, "withdraw failed");
    }
}
```

**Generating the Merkle tree off-chain (TypeScript)**:

```typescript
import { keccak256, encodePacked } from "viem";
import { MerkleTree } from "merkletreejs";

const whitelist: `0x${string}`[] = [
  "0xAlice...",
  "0xBob...",
  "0xCharlie...",
];

// Create leaves
const leaves = whitelist.map((addr) =>
  keccak256(encodePacked(["address"], [addr]))
);

// Build tree
const tree = new MerkleTree(leaves, keccak256, { sortPairs: true });
const root = tree.getHexRoot();
console.log("Merkle root:", root);

// Get proof for a specific address
const leaf = keccak256(encodePacked(["address"], ["0xAlice..."]));
const proof = tree.getHexProof(leaf);
console.log("Proof for Alice:", proof);
```

## Reveal Mechanism

### Hidden Metadata + Reveal

Before reveal, all tokens return a placeholder. After reveal, they return the real metadata.

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract RevealableNFT is ERC721, Ownable {
    string public hiddenMetadataURI;
    string public baseURI;
    bool public revealed;
    string public provenanceHash; // SHA256 of concatenated metadata hashes

    uint256 private _nextTokenId;

    constructor(
        string memory name,
        string memory symbol,
        string memory _hiddenURI,
        string memory _provenanceHash
    ) ERC721(name, symbol) Ownable(msg.sender) {
        hiddenMetadataURI = _hiddenURI;
        provenanceHash = _provenanceHash;
    }

    function reveal(string memory _baseURI) external onlyOwner {
        require(!revealed, "already revealed");
        baseURI = _baseURI;
        revealed = true;
    }

    function tokenURI(uint256 tokenId) public view override returns (string memory) {
        _requireOwned(tokenId);

        if (!revealed) {
            return hiddenMetadataURI;
        }

        return string(abi.encodePacked(baseURI, Strings.toString(tokenId)));
    }

    function mint() external payable {
        _safeMint(msg.sender, _nextTokenId++);
    }
}
```

### Provenance Hash

Publish the provenance hash before minting begins. This proves the metadata was not changed after minting.

```typescript
import { createHash } from "crypto";

// Compute provenance hash from all metadata files
function computeProvenanceHash(metadataHashes: string[]): string {
  const concatenated = metadataHashes.join("");
  return createHash("sha256").update(concatenated).digest("hex");
}
```

## Royalties (ERC-2981)

On-chain royalty info that marketplaces can query.

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/token/common/ERC2981.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract RoyaltyNFT is ERC721, ERC2981, Ownable {
    uint256 private _nextTokenId;

    constructor(
        string memory name,
        string memory symbol,
        address royaltyReceiver,
        uint96 royaltyBps // e.g. 500 = 5%
    ) ERC721(name, symbol) Ownable(msg.sender) {
        // Set default royalty for all tokens
        _setDefaultRoyalty(royaltyReceiver, royaltyBps);
    }

    function mint(address to) external onlyOwner {
        _safeMint(to, _nextTokenId++);
    }

    /// @notice Override royalty for a specific token
    function setTokenRoyalty(
        uint256 tokenId,
        address receiver,
        uint96 feeNumerator
    ) external onlyOwner {
        _setTokenRoyalty(tokenId, receiver, feeNumerator);
    }

    function supportsInterface(bytes4 interfaceId)
        public view override(ERC721, ERC2981) returns (bool)
    {
        return super.supportsInterface(interfaceId);
    }
}
```

## Metadata

### On-Chain vs Off-Chain

| Approach | Pros | Cons |
| --- | --- | --- |
| **On-chain** (base64 SVG) | Fully decentralized, permanent | High gas cost, limited complexity |
| **IPFS** | Decentralized, content-addressed | Needs pinning service |
| **Arweave** | Permanent storage | Higher initial cost |
| **Centralized (S3/CDN)** | Cheap, fast, flexible | Single point of failure, mutable |

## Complete Template

Full NFT collection with whitelist, public mint, reveal, and royalties.

### Solidity Contract

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/token/common/ERC2981.sol";
import "@openzeppelin/contracts/utils/cryptography/MerkleProof.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/utils/Strings.sol";

/// @title NFTCollection
/// @notice Full-featured NFT: whitelist, public mint, reveal, royalties.
contract NFTCollection is ERC721, ERC2981, Ownable {
    using Strings for uint256;

    uint256 public constant MAX_SUPPLY = 10000;
    uint256 public constant WL_PRICE = 0.03 ether;
    uint256 public constant PUBLIC_PRICE = 0.05 ether;
    uint256 public constant MAX_PER_TX = 5;

    uint256 private _nextTokenId;
    bytes32 public merkleRoot;
    string public hiddenURI;
    string public baseURI;
    bool public revealed;

    enum Phase { Closed, Whitelist, Public }
    Phase public phase;

    mapping(address => uint256) public mintedPerWallet;

    event PhaseChanged(Phase newPhase);
    event Revealed(string baseURI);

    constructor(
        string memory name,
        string memory symbol,
        string memory _hiddenURI,
        bytes32 _merkleRoot,
        address royaltyReceiver,
        uint96 royaltyBps
    ) ERC721(name, symbol) Ownable(msg.sender) {
        hiddenURI = _hiddenURI;
        merkleRoot = _merkleRoot;
        _setDefaultRoyalty(royaltyReceiver, royaltyBps);
    }

    // ── Minting ────────────────────────────────────────────
    function whitelistMint(uint256 qty, bytes32[] calldata proof) external payable {
        require(phase == Phase.Whitelist, "WL closed");
        require(qty > 0 && qty <= MAX_PER_TX, "bad qty");
        require(_nextTokenId + qty <= MAX_SUPPLY, "sold out");
        require(msg.value >= WL_PRICE * qty, "underpaid");

        bytes32 leaf = keccak256(abi.encodePacked(msg.sender));
        require(MerkleProof.verify(proof, merkleRoot, leaf), "bad proof");

        for (uint256 i = 0; i < qty; i++) {
            _safeMint(msg.sender, _nextTokenId++);
        }
    }

    function publicMint(uint256 qty) external payable {
        require(phase == Phase.Public, "public closed");
        require(qty > 0 && qty <= MAX_PER_TX, "bad qty");
        require(_nextTokenId + qty <= MAX_SUPPLY, "sold out");
        require(msg.value >= PUBLIC_PRICE * qty, "underpaid");
        require(mintedPerWallet[msg.sender] + qty <= MAX_PER_TX, "wallet limit");

        mintedPerWallet[msg.sender] += qty;

        for (uint256 i = 0; i < qty; i++) {
            _safeMint(msg.sender, _nextTokenId++);
        }
    }

    // ── Reveal ─────────────────────────────────────────────
    function reveal(string memory _baseURI) external onlyOwner {
        require(!revealed, "already revealed");
        baseURI = _baseURI;
        revealed = true;
        emit Revealed(_baseURI);
    }

    function tokenURI(uint256 tokenId) public view override returns (string memory) {
        _requireOwned(tokenId);
        if (!revealed) return hiddenURI;
        return string(abi.encodePacked(baseURI, tokenId.toString()));
    }

    // ── Admin ──────────────────────────────────────────────
    function setPhase(Phase _phase) external onlyOwner {
        phase = _phase;
        emit PhaseChanged(_phase);
    }

    function setMerkleRoot(bytes32 _root) external onlyOwner {
        merkleRoot = _root;
    }

    function withdraw() external onlyOwner {
        (bool ok, ) = owner().call{value: address(this).balance}("");
        require(ok, "withdraw failed");
    }

    function totalSupply() public view returns (uint256) {
        return _nextTokenId;
    }

    function supportsInterface(bytes4 interfaceId)
        public view override(ERC721, ERC2981) returns (bool)
    {
        return super.supportsInterface(interfaceId);
    }
}
```

### Hardhat Test

```typescript
import { expect } from "chai";
import { ethers } from "hardhat";
import { loadFixture } from "@nomicfoundation/hardhat-toolbox/network-helpers";
import { MerkleTree } from "merkletreejs";
import { keccak256, solidityPackedKeccak256 } from "ethers";

describe("NFTCollection", function () {
  async function deployFixture() {
    const [owner, alice, bob, royaltyWallet] = await ethers.getSigners();

    // Build whitelist Merkle tree
    const whitelist = [alice.address, bob.address];
    const leaves = whitelist.map((addr) =>
      solidityPackedKeccak256(["address"], [addr])
    );
    const tree = new MerkleTree(leaves, keccak256, { sortPairs: true });
    const root = tree.getHexRoot();

    const NFT = await ethers.getContractFactory("NFTCollection");
    const nft = await NFT.deploy(
      "TestNFT",
      "TNFT",
      "ipfs://hidden",
      root,
      royaltyWallet.address,
      500 // 5% royalty
    );

    return { nft, owner, alice, bob, royaltyWallet, tree };
  }

  it("whitelist mint with valid proof", async function () {
    const { nft, owner, alice, tree } = await loadFixture(deployFixture);

    await nft.setPhase(1); // Whitelist

    const leaf = solidityPackedKeccak256(["address"], [alice.address]);
    const proof = tree.getHexProof(leaf);

    await nft.connect(alice).whitelistMint(1, proof, {
      value: ethers.parseEther("0.03"),
    });

    expect(await nft.ownerOf(0)).to.equal(alice.address);
  });

  it("rejects invalid whitelist proof", async function () {
    const { nft, owner, tree } = await loadFixture(deployFixture);
    const [, , , , stranger] = await ethers.getSigners();

    await nft.setPhase(1);

    const fakeLeaf = solidityPackedKeccak256(["address"], [stranger.address]);
    const fakeProof = tree.getHexProof(fakeLeaf);

    await expect(
      nft.connect(stranger).whitelistMint(1, fakeProof, {
        value: ethers.parseEther("0.03"),
      })
    ).to.be.revertedWith("bad proof");
  });

  it("public mint and reveal", async function () {
    const { nft, owner, alice } = await loadFixture(deployFixture);

    await nft.setPhase(2); // Public

    await nft.connect(alice).publicMint(2, {
      value: ethers.parseEther("0.1"),
    });

    // Before reveal
    expect(await nft.tokenURI(0)).to.equal("ipfs://hidden");

    // Reveal
    await nft.reveal("ipfs://QmRevealed/");
    expect(await nft.tokenURI(0)).to.equal("ipfs://QmRevealed/0");
    expect(await nft.tokenURI(1)).to.equal("ipfs://QmRevealed/1");
  });

  it("enforces royalty info", async function () {
    const { nft, royaltyWallet } = await loadFixture(deployFixture);
    const salePrice = ethers.parseEther("1");
    const [receiver, amount] = await nft.royaltyInfo(0, salePrice);

    expect(receiver).to.equal(royaltyWallet.address);
    expect(amount).to.equal(ethers.parseEther("0.05")); // 5%
  });
});
```

### Foundry Test

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "forge-std/Test.sol";
import "../src/NFTCollection.sol";

contract NFTCollectionTest is Test {
    NFTCollection nft;
    address owner = address(this);
    address alice = makeAddr("alice");
    address bob = makeAddr("bob");
    address royaltyWallet = makeAddr("royalty");

    // For simplicity, use a dummy merkle root (real tests should build the tree)
    bytes32 merkleRoot;

    function setUp() public {
        // Compute merkle root for [alice, bob]
        bytes32 leafAlice = keccak256(abi.encodePacked(alice));
        bytes32 leafBob = keccak256(abi.encodePacked(bob));
        // For 2 leaves, root = hash(min, max)
        merkleRoot = leafAlice < leafBob
            ? keccak256(abi.encodePacked(leafAlice, leafBob))
            : keccak256(abi.encodePacked(leafBob, leafAlice));

        nft = new NFTCollection(
            "TestNFT",
            "TNFT",
            "ipfs://hidden",
            merkleRoot,
            royaltyWallet,
            500
        );
    }

    function test_PublicMint() public {
        nft.setPhase(NFTCollection.Phase.Public);

        vm.deal(alice, 1 ether);
        vm.prank(alice);
        nft.publicMint{value: 0.1 ether}(2);

        assertEq(nft.ownerOf(0), alice);
        assertEq(nft.ownerOf(1), alice);
        assertEq(nft.totalSupply(), 2);
    }

    function test_Reveal() public {
        nft.setPhase(NFTCollection.Phase.Public);

        vm.deal(alice, 1 ether);
        vm.prank(alice);
        nft.publicMint{value: 0.05 ether}(1);

        // Before reveal
        assertEq(nft.tokenURI(0), "ipfs://hidden");

        // After reveal
        nft.reveal("ipfs://QmRevealed/");
        assertEq(nft.tokenURI(0), "ipfs://QmRevealed/0");
    }

    function test_Royalty() public view {
        (address receiver, uint256 amount) = nft.royaltyInfo(0, 1 ether);
        assertEq(receiver, royaltyWallet);
        assertEq(amount, 0.05 ether); // 5%
    }

    function test_RevertWhenSoldOut() public {
        // This would take too long for 10k, so just test the check
        nft.setPhase(NFTCollection.Phase.Public);

        vm.deal(alice, 100 ether);
        vm.prank(alice);
        vm.expectRevert("bad qty");
        nft.publicMint{value: 0.3 ether}(6); // MAX_PER_TX = 5
    }

    function test_Withdraw() public {
        nft.setPhase(NFTCollection.Phase.Public);

        vm.deal(alice, 1 ether);
        vm.prank(alice);
        nft.publicMint{value: 0.05 ether}(1);

        uint256 balBefore = owner.balance;
        nft.withdraw();
        assertEq(owner.balance, balBefore + 0.05 ether);
    }
}
```
